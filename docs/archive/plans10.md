## API Design

### Basic Usage

```typescript
import { animate } from '@animate/core';

// Simple animation
animate('.element', {
  opacity: [0, 1],
  translateY: [-20, 0]
}, {
  duration: 1000,
  easing: 'easeOutCubic'
});
```

### Timeline Usage

```typescript
import { timeline } from '@animate/core';

const tl = timeline();

tl.add('.element1', { scale: 1.5 }, { duration: 500 })
  .add('.element2', { rotate: 360 }, { duration: 1000 }, '-=250')
  .add('.element3', { opacity: 0 }, { duration: 500 });

tl.play();
```

### Spring Animations

```typescript
import { spring } from '@animate/physics';

spring('.element', {
  translateX: 200
}, {
  stiffness: 100,
  damping: 10,
  mass: 1
});
```

### Scroll Animations

```typescript
import { scrollAnimate } from '@animate/scroll';

scrollAnimate('.element', {
  opacity: [0, 1],
  translateY: [100, 0]
}, {
  trigger: '.element',
  start: 'top 80%',
  end: 'top 20%',
  scrub: true
});
```
